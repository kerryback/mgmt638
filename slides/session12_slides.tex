\begin{frame}{Automated Scheduled Trading}
\begin{enumerate}
\item Download data at specified time of day (e.g., 30 minutes before market close)
\item Run model to generate trading signals
\item Run trading protocol to convert signals and current holdings to desired trades
\item Connect to broker API and submit trades
\item Receive execution confirmations
\end{enumerate}
\end{frame}

\begin{frame}{Real-Time Monitoring}
\begin{itemize}
\item Stream live price data from broker
\item Trigger alarms on extreme price movements
\item Maybe trigger stop-loss orders
\end{itemize}
\end{frame}

\begin{frame}{Introduction to Alpaca-py}
\begin{itemize}
\item \texttt{pip install alpaca-py}
\item Sign up for free account at \href{https://alpaca.markets}{alpaca.markets}
\item Create API key and secret key
\item Crypto market data requires no authentication
\item Stock market data and trading require API keys
\item Sandbox/paper trading environment available for testing
\end{itemize}
\end{frame}

\begin{frame}{Core Features and APIs}
\textbf{Three Main API Products:}
\begin{enumerate}
\item \textbf{Market Data API}
\begin{itemize}
\item Live and historical data for 5,000+ stocks, 20+ cryptocurrencies, and options
\item Stock and crypto bar data, quotes, trades, and news
\end{itemize}
\item \textbf{Trading API}
\begin{itemize}
\item Execute trades in stocks, crypto, and options
\item Order management and position tracking
\end{itemize}
\item \textbf{Broker API}
\begin{itemize}
\item Build investment applications
\item Platform for robo-advisors or brokerage services
\end{itemize}
\end{enumerate}
\end{frame}
